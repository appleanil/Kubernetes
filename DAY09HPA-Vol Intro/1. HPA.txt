pod scaling in Kubernetes
=========================

--> Increase/decrease the pods manually is called a manual scaling [replicas 2 to 3, 3 to 4, 4 to 2]

--> HPA (Horizontal Pod Autoscaler) in Kubernetes automatically increases or decreases the number of Pods in a Deployment, ReplicaSet, or StatefulSet based on
    metrics like CPU, memory, or custom metrics.

--> When we are creating the HorizonatalPodAutoScale background it will automatically created service [if we define], ReplicaSets, Deployment and hpa beacause of hpa

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: php-apache
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: php-apache
  minReplicas: 1
  maxReplicas: 10
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 80

#deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: hpadeployment              #name should be same in hpa scaleTargetRef metadata
  labels:
    name: hpadeployment
spec:
  replicas: 1                      #just reference, its optainal also and not mandatory and first priority is hpa replicas only
  selector:
    matchLabels:
      name: hpapod
  template:
    metadata:
      labels:
        name: hpapod
    spec:
      containers:
      - name: hpacontainer
        image: registry.k8s.io/hpa-example
        ports:
        - name: http
          containerPort: 80           #port should be same in service port also
        resources:
          requests:
            cpu: "100m"
            memory: "64Mi"
          limits:
            cpu: "100m"
            memory: "128Mi"
---
#HPA

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: hpadeploymentautoscaler
spec:
  scaleTargetRef:
    apiVersion: apps/v1      #This section refer to above deployment and should be same "name hpdeployment"
    kind: Deployment
    name: hpadeployment      #name should be same in deploy pod
  minReplicas: 2             # first priority hpa, controll should be in hpa only, that deployment replicas 1 is just for formalty
  maxReplicas: 4
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 30      #in realtime it should be 80,85
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80

---
