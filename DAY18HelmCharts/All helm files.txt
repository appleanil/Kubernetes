1. helm upgrade --install javawebapp ./javawebapp --dry-run --debug ------> # shows all yaml files install/upgrade without applying (best for testing templates)

2. helm install javawebapp ./javawebapp -n prod --create-namespace -------> # Install chart into 'prod' namespace (creates ns if not exists)

3. helm ls -n prod -------------------------------------------------------> # List all releases in 'prod' namespace

4. helm lint javawebapp --------------------------------------------------> # Validate chart structure & syntax

5. helm package javawebapp -----------------------------------------------> # Creates packaged chart (javawebapp-1.0.0.tgz)

6. helm upgrade --install javawebapp2 javawebapp-1.0.0.tgz -n prod -------> # Deploy NEW release from packaged chart into 'prod'

7. helm upgrade --install javawebapp2 javawebapp-1.0.0.tgz ---------------> # Deploy into default namespace

8. helm ls -A
   # List releases across all namespaces

9. kubectl get all -n prod
   # View all resources in prod namespace

10. kubectl get all -n default
    # View resources in default namespace

11. helm show values javawebapp-1.0.0.tgz
    # Display default values.yaml from packaged chart

12. helm template ./javawebapp > output.yaml
    # Render Kubernetes YAML (without deploying)

13. kubectl apply -f output.yaml
    # Apply rendered YAML manually

14. helm status javawebapp --show-resources -n prod
    # Show release status + all created resources

15. helm uninstall javawebapp -n prod
    # Delete Helm release

16. helm history javawebapp -n prod
    # Show revision history

17. helm rollback javawebapp 1 -n prod
    # Rollback to revision 1

simple with NF-Server
----------------------

kind: PersistentVolume
metadata:
  name: {{ .Release.Name }}-mongo-pv
spec:
  capacity:
    storage: {{ .Values.storage.size }}
  accessModes:
    - ReadWriteMany   # NFS supports multi-node access
  persistentVolumeReclaimPolicy: Retain    #from here to last line this is difference
  nfs:
    server: {{ .Values.nfs.server }}
    path: {{ .Values.nfs.path }}

7. mongo.yaml with-out NFS-Server
---------------------------------
apiVersion: apps/v1
kind: Deployment
metadata:
  name: {{ .Release.Name }}-mongo
  namespace: {{ .Values.namespace }}
  labels:
    app: {{ .Release.Name }}-mongo
spec:
  replicas: 1
  selector:
    matchLabels:
      app: {{ .Release.Name }}-mongo
  template:
    metadata:
      labels:
        app: {{ .Release.Name }}-mongo
    spec:
      containers:
      - name: mongo
        image: {{ .Values.mongo.image }}
        ports:
        - containerPort: {{ .Values.mongo.port }}

        env:
        - name: MONGO_INITDB_ROOT_USERNAME
          valueFrom:
            secretKeyRef:
              name: {{ .Release.Name }}-secret
              key: DB_USERNAME

        - name: MONGO_INITDB_ROOT_PASSWORD
          valueFrom:
            secretKeyRef:
              name: {{ .Release.Name }}-secret
              key: DB_PASSWORD

        volumeMounts:
        - name: mongo-storage
          mountPath: /data/db

      volumes:
      - name: mongo-storage
        persistentVolumeClaim:
          claimName: {{ .Release.Name }}-pvc
---
apiVersion: v1
kind: Service
metadata:
  name: {{ .Release.Name }}-mongo-svc
  namespace: {{ .Values.namespace }}
spec:
  selector:
    app: {{ .Release.Name }}-mongo
  ports:
    - port: {{ .Values.mongo.port }}
      targetPort: {{ .Values.mongo.port }}
  type: ClusterIP

1. Basic Helm Commands
----------------------
helm version ------------------> Shows the installed Helm version

helm help ---------------------> Displays help for Helm commands

helm repo list ----------------> Lists all added repositories (shows repo URLs)

helm repo add bitnami https://charts.bitnami.com/bitnami   # Adds Bitnami repository

helm repo update --------------> Updates repo index (fetch latest charts)

helm search repo nginx --------> Searches charts in added repositories

helm search hub nginx ---------> Searches charts in Artifact Hub (online)


2. Working with Charts
-----------------------
helm create mychart ----------------------> Creates a new Helm chart with default structure

helm lint mychart ------------------------> Validates the chart for syntax and structural errors

helm package mychart ---------------------> Packages the chart into a .tgz file

helm show chart mychart ------------------> Displays metadata from Chart.yaml

helm show values mychart ----------------> Shows default values from values.yaml

helm show all mychart -------------------> Displays all chart information

helm template myapp ./mychart -----------> Renders Kubernetes YAML without deploying


3. Install / Upgrade / Uninstall
---------------------------------
helm install myapp ./mychart
helm install myapp bitnami/nginx
helm install myapp ./mychart -f values.yaml
helm install myapp ./mychart --set replicaCount=3

helm upgrade myapp ./mychart
helm upgrade myapp ./mychart -f values.yaml
helm upgrade myapp ./mychart --set image.tag=2.0
helm uninstall myapp


4. Release Management
---------------------
helm list -------------------------------> application info name, namespace, revision, chartname, status, decription and updated timings

helm list -A ----------------------------> all application info name, namespace, revision, chartname, status, decription and updated timings

helm list -n prod -----------------------> application info name, namespace, revision, chartname, status, decription and updated timings

helm status myapp -n prod ---------------> application info name, namespace, deployed revision, last deployed

helm history myapp -n prod --------------> application info revision, chartname, status, decription and updated timings


5. Rollback & Debugging
-----------------------
helm rollback myapp 1
helm rollback myapp 2

helm get all myapp -n prod ----------------------> shows all yaml files chart.yaml, values.yaml and template folder deploy.yaml.service.yaml,hpa.yaml.....
helm get values myapp -n prod -------------------> repliaCount show u
helm get manifest myapp -n prod -----------------> shows template folder deploy.yaml.service.yaml,hpa.yaml.....

helm install myapp ./mychart --dry-run ----------> check all values are correct or not
helm install myapp ./mychart --debug ------------> before using --dry-run after use this bcz is shows errors and logs files
