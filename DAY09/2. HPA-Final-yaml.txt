#deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: hpadeployment
  namespace: prod
  labels:
    app: hpadeployment
    env: prod
spec:
  replicas: 1   #just reference, its optainals also and not mandatory and first priority is hpa replicas only
  selector:
    matchLabels:
      name: hpapod
  template:
    metadata:
      labels:
        name: hpapod
    spec:
      containers:
      - name: hpacontainer
        image: registry.k8s.io/hpa-example
        ports:
        - name: http
          containerPort: 80
        resources:
          requests:
            cpu: "100m"
            memory: "64Mi"
          limits:
            cpu: "100m"
            memory: "128Mi"
---
#HPA

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: hpadeploymentautoscaler
  namespace: prod
spec:
  scaleTargetRef:
    apiVersion: apps/v1 #This section refer to above deployment and should be same "name hpdeployment"
    kind: Deployment
    name: hpadeployment
  minReplicas: 2   # first priority hpa, controll should be in hpa only, that deployment replicas 1 is just for formal
  maxReplicas: 4
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 30  #in realtime it should be 80,85
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80

---
